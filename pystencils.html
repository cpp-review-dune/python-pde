import numpy as np
import pystencils as ps
import sympy as sp
from pystencils import create_kernel
from pystencils import plot as plt

sp.init_printing()

input_arr = np.random.rand(1024, 1024)
output_arr = np.zeros_like(input_arr)

def numpy_kernel():
    output_arr[1:-1, 1:-1] = (
        input_arr[2:, 1:-1]
        + input_arr[:-2, 1:-1]
        + input_arr[1:-1, 2:]
        + input_arr[1:-1, :-2]
    )

%%timeit
numpy_kernel()

)

symbolic_description
